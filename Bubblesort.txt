//bubble sort array
#include <iostream>
using namespace std;

int main()
{
    int n;
    
    cout << "Numar elemente array : ";
    cin >> n;
    
    int array[n];
    
    cout << "Introduceti elementele array-ului : ";
    
    for(int i = 0; i < n; i++)
    {
        cin >> array[i];
    }
    
    
    cout << "Array initial" <<endl;
    
    for(int i : array)
    {
        cout << i << " ";
    }
    
    cout << endl;
    
    //bubblesort in ordine crescatoare
    
    for(int i = 0; i < n - 1; i++)
    {
        for(int j = 0; j < n - i - 1; j++)
        {
            if(array[j] > array[j+1])
            {
                int aux  = array[j];
                array[j] = array[j+1];
                array[j+1] = aux;
            }
        }
    }
    
    cout << "Array ordonat crescator prin bubblesort" << endl;
    
    for(int i : array)
    {
        cout << i << " ";
    }
    
    return 0;
}
